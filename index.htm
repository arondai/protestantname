        dom.startScreen.classList.add('hidden');

        dom.questionScreen.classList.remove('hidden');

        dom.questionScreen.classList.add('fade-in');

    function showQuestion() {

        const questions = langData[currentLang].questions;

        const question = questions[currentQuestionIndex];

        dom.questionText.textContent = question.q;

        dom.questionBtns[0].textContent = question.a[0].text;

        dom.questionBtns[0].dataset.type = question.a[0].type;

        dom.questionBtns[1].textContent = question.a[1].text;

        dom.questionBtns[1].dataset.type = question.a[1].type;

        const progress = ((currentQuestionIndex + 1) / questions.length) * 100;

        dom.progressBar.style.width = `${progress}%`;

    }


    function selectAnswer(type) {

        mbtiScores[type]++;

        currentQuestionIndex++;

        if (currentQuestionIndex < langData[currentLang].questions.length) {

            showQuestion();

        } else {

            showLoading();

        }

    }


    function showLoading() {

        dom.questionScreen.classList.add('hidden');

        dom.loadingScreen.classList.remove('hidden');

        setTimeout(calculateResult, 2000);

    }


    function calculateResult() {

        let result = "";

        result += mbtiScores.E > mbtiScores.I ? "E" : "I";

        result += mbtiScores.S > mbtiScores.N ? "S" : "N";

        result += mbtiScores.T > mbtiScores.F ? "T" : "F";

        result += mbtiScores.J > mbtiScores.P ? "J" : "P";

        findSoulName(result);

    }


    function findSoulName(mbtiType) {

        const namePool = biblicalNamesDatabase.filter(name => name.associatedMBTI.includes(mbtiType));

        let chosenName;

        if (namePool.length > 0) {

            chosenName = namePool[Math.floor(Math.random() * namePool.length)];

        } else {

            chosenName = { id: 0, koreanName: "은총", originalName: "기본값", associatedMBTI: [] };

            if (!langData.ko.biblicalNames.find(n => n.id === 0)) {

                langData.ko.biblicalNames.push({ id: 0, reason: "어떤 상황에서도 믿음을 지키는 당신의 모습에서 새로운 시작의 가능성을 봅니다." });

                langData.en.biblicalNames.push({ id: 0, reason: "In your unwavering faith in any situation, we see the possibility of a new beginning." });

                langData.es.biblicalNames.push({ id: 0, reason: "En tu fe inquebrantable en cualquier situación, vemos la posibilidad de un nuevo comienzo." });

            }

        }

        displayResult(mbtiType, chosenName);

    }


    function displayResult(mbtiType, soulNameData) {

        const translatedReasonObj = langData[currentLang].biblicalNames.find(n => n.id === soulNameData.id);

        dom.loadingScreen.classList.add('hidden');

    }
